Why Scala?

  • Expressive
    • First class functions
  • Concise
    • Type inference
    • Literal syntax for function creation
  • Java interoperability
    • Can reuse Java libraries
    • Can reuse Java tools
    • No performance penalty

How Scala?

  • Compiles to Java bytecode
  • Works with any standard JVM
    • Or even some non-standard JVMs like Dalvik
    • Scala compiler written by one of the authors of the Java compiler

A simple class in Java


public class Car {
    private int year;
    private String model;

    Car(int year, String model) {
        this.year = year;
        this.model = model;
    }

    int getYear() {
        return year;
    }

    String getModel() {
        return model;
    }

    boolean equals(Object obj) {
        if (obj == null || !(obj instanceof Car)) {
            return false;
        }
        Car car = (Car) obj;
        return year == car.year && model == car.model;
     }
}

A simple class in Scala


class Car(_year: Int, _model: String) {
    private var year: Int = _year
    private var model: String = _model

    def getYear: Int = year

    def getModel: String = model

    override def equals(obj: Any) {
        if (obj == null || !(obj.isInstanceOf[Car])) {
            false
        } else {
            val car = obj.asInstanceOf[Car]
            year == car.year && model == car.model
        }
     }
}
